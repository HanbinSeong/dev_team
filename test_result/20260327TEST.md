```python
# fib.py
"""Core Fibonacci implementation.

Public API:
- fib(n: int) -> int
  Returns the nth Fibonacci number with F(0) = 0, F(1) = 1.

Input validation:
- n must be an int; otherwise raise TypeError("n must be an integer").
- n must be non-negative; otherwise raise ValueError("n must be non-negative").

Algorithm:
- Iterative, O(n) time, O(1) space.
- Uses a = 0, b = 1 and iterates n times updating (a, b) = (b, a + b).
"""
from __future__ import annotations


def fib(n: int) -> int:
    """Return the nth Fibonacci number using an iterative approach.

    The sequence is defined as:
      F(0) = 0, F(1) = 1, F(n) = F(n-1) + F(n-2) for n >= 2.

    Args:
        n (int): Non-negative index of the Fibonacci sequence.

    Returns:
        int: The nth Fibonacci number.

    Raises:
        TypeError: If n is not an integer.
        ValueError: If n is negative.
    """
    if not isinstance(n, int):
        raise TypeError("n must be an integer")
    if n < 0:
        raise ValueError("n must be non-negative")

    a, b = 0, 1
    for _ in range(n):
        a, b = b, a + b
    return a


__all__ = ["fib"]
```

```python
# word_top5.py

#!/usr/bin/env python3
"""
Word Top5 - A small utility to compute top-N words from a text file.

Features:
- Reads text from a file with encoding fallbacks (UTF-8 first, then common encodings).
- Tokenizes using a robust regex that preserves contractions (e.g., don't) as single tokens.
- Normalizes to lowercase.
- Computes top-N words with a stable tie-breaker (lexicographic).
- Produces a JSON result with metadata and the list of top words.
- Exposes a CLI and can be imported as a library.

Note: This module uses only the Python standard library.
"""

from __future__ import annotations

import argparse
import datetime
import json
import re
from collections import Counter
from pathlib import Path
from typing import Dict, List, Optional


# Public API

def read_text_from_file(input_path: str, preferred_encodings: Optional[List[str]] = None) -> str:
    """Read text from file with encoding fallbacks.

    - Tries UTF-8 first, then a set of common encodings.
    - Returns decoded text on success.
    - Raises FileNotFoundError if path is invalid.
    - Raises UnicodeDecodeError (or a wrapped one) if decoding fails for all encodings.
    """
    path = Path(input_path)
    if not path.exists():
        raise FileNotFoundError(f"Input file not found: {input_path}")

    # Read raw bytes first to try multiple decodings
    raw: bytes
    with open(path, "rb") as f:
        raw = f.read()

    encodings = preferred_encodings or ["utf-8", "utf-8-sig", "latin-1", "cp1252"]
    last_err: Optional[Exception] = None
    for enc in encodings:
        try:
            text = raw.decode(enc)
            return text
        except UnicodeDecodeError as e:
            last_err = e
            continue

    # If we reach here, decoding failed for all encodings
    if last_err is not None:
        raise UnicodeDecodeError(
            last_err.encoding,
            last_err.object,
            last_err.start,
            last_err.end,
            last_err.reason,
        )
    raise UnicodeDecodeError("unknown", raw, 0, 1, "Failed to decode with provided encodings")


def tokenize(text: str) -> List[str]:
    """Tokenize text into words, preserving internal apostrophes (e.g., don't)."""
    if text is None:
        return []
    lowered = text.lower()
    # Regex preserves words with internal apostrophes, like don't, it's, won't
    tokens = re.findall(r"[A-Za-z]+(?:'[A-Za-z]+)*", lowered)
    return [t for t in tokens if t]


def top_n_words(words: List[str], n: int = 5) -> List[Dict[str, int]]:
    """Return top-n words as a list of dicts with keys 'word' and 'count'.

    - Words are expected to be pre-normalized (lowercase).
    - Sorting is by (-count, word) to ensure deterministic tie-breaking.
    - If there are fewer than n unique words, returns as many as exist.
    """
    if n < 1:
        raise ValueError("n must be >= 1")

    counts = Counter(words)
    # Sort by count desc, then by word asc for deterministic order
    items = sorted(counts.items(), key=lambda kv: (-kv[1], kv[0]))
    top = items[:n]
    return [{"word": w, "count": c} for w, c in top]


def save_result_to_json(result: Dict, output_path: str) -> None:
    """Save a dict to a pretty-printed JSON file ( UTF-8, ensure dirs )."""
    path = Path(output_path)
    path.parent.mkdir(parents=True, exist_ok=True)
    with open(path, "w", encoding="utf-8") as f:
        json.dump(result, f, ensure_ascii=False, indent=2)


def process_file(input_path: str, output_path: str = "result.json", top_n: int = 5) -> Dict:
    """Orchestrate reading, tokenizing, counting, and saving results.

    Returns the result dictionary which can be used programmatically.
    """
    # 1) Read
    text = read_text_from_file(input_path)
    if text is None or text.strip() == "":
        raise ValueError("Empty text")

    # 2) Tokenize
    words = tokenize(text)

    # 3) Top-N
    top_words = top_n_words(words, n=top_n)

    # 4) Build result structure
    result = {
        "source": str(input_path),
        "top_n": top_n,
        "generated_at": datetime.datetime.now().isoformat(sep="T", timespec="seconds"),
        "top_words": top_words,
    }

    # 5) Persist
    save_result_to_json(result, output_path)

    return result


def _build_cli_parser() -> argparse.ArgumentParser:
    parser = argparse.ArgumentParser(description="Compute top-N words from a text file and export as JSON.")
    parser.add_argument("-i", "--input", dest="input_path", required=True, help="Path to input text file")
    parser.add_argument("-o", "--output", dest="output_path", default="result.json", help="Path to output JSON file")
    parser.add_argument("-n", "--top", dest="top_n", type=int, default=5, help="Number of top words to compute (default: 5)")
    parser.add_argument("--quiet", action="store_true", help="Suppress non-error messages")
    return parser


def main() -> None:
    parser = _build_cli_parser()
    args = parser.parse_args()

    try:
        result = process_file(args.input_path, args.output_path, args.top_n)
        if not args.quiet:
            print(f"Top-{args.top_n} words saved to: {args.output_path}")
    except Exception as exc:  # pragma: no cover - CLI error path
        import sys
        print(f"Error: {exc}", file=sys.stderr)
        sys.exit(1)


if __name__ == "__main__":
    main()
```
